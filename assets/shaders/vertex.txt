#version 450 core

layout(location = 0) in vec3 aPosition;
layout(location = 1) in vec2 aTexCoord;

out vec2 vTexCoord;

layout(std430, binding = 0) buffer Transforms
{
    mat4 models[];
};

uniform mat4 uVP;

void main()
{
    vec4 worldPosition = uVP * models[gl_InstanceID] * vec4(aPosition, 1.0);
    vTexCoord = aTexCoord;
    gl_Position = worldPosition;
}
